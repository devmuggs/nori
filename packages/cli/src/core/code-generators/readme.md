# Nori - Code Generators

This directory contains code generator feature for Nori. Code generators take a NoriState and produce code to help users interact with i18n in their projects.

## Available Code Generators

-   TypeScript Code Generator: Generates TypeScript code to facilitate i18n integration.

## Example Input nori.yaml

```yaml
# Nori Configuration File
preferredLocale: en-GB
author: tristan
projectName: cli
projectDescription: A Nori project
version: 1.0.0

# Add your collections and entries below
collections:
    client:
        description: "Client-specific messages."
        welcome-to-nori:
            description: "Greeting message for new users on home page."
            params:
                username
                    type: string
                    description: "The name of the user."
            locales:
                en-GB: "Welcome to Nori, {{username}}!"
                ja-JP: "ノリへようこそ、{{username}}さん！"
        good-morning:
            description: "Morning greeting message."
            locales:
                en-GB: "Good morning!"
                ja-JP: "おはようございます！"

entries:
    lets-get-started:
        description: "Displayed on first step of onboarding wizard."
        params:
            topic:
                type: string
                description: "The topic to get started with."
        locales:
            en-GB: "Let's get started with {{topic}}."
            ja-JP: "さあ、{{topic}}を始めましょう！"

```

## Example Output Generated Code

```typescript
// TypeScript code generated by Nori Code Generator

import { NoriLocale, NoriI18nCollection } from "../cli/src/core/index.js";

export const nori = {
	/** Client-specific messages. */
	client: {
		welcomeToNori: (params: {
			/** The name of the user. */
			username: string | undefined;
		}): NoriI18nCollection => {
			const { username = "guest" } = params;
			return {
				[NoriLocale.EnglishBritish]: `Welcome to Nori, ${username}!`,
				[NoriLocale.Japanese]: `ノリへようこそ、${username}さん！`
			};
		},
		goodMorning: (): NoriI18nCollection => {
			return {
				[NoriLocale.EnglishBritish]: `Good morning!`,
				[NoriLocale.Japanese]: `おはようございます！`
			};
		}
	} as const
} as const;

export default nori;
```
